│ EmergencyManualTrading│  ──────┐
│  (紧急手动交易)       │        │
└──────────────────────┘        │
                                │ emit('orderExecuted')
                                │
                                ▼
                        ┌──────────────────┐
                        │ TradingDashboard │
                        │   (父组件)       │
                        └──────────────────┘
                                │
                                │ ref.fetchRecentOrders()
                                │
                                ▼
                        ┌──────────────────────┐
                        │ RecentTradingRecords │
                        │  (最近交易记录)       │
                        └──────────────────────┘
                                │
                                │ watch(lastMessage)
                                │
                                ▼
                        ┌──────────────────┐
                        │  WebSocket       │

                                │
